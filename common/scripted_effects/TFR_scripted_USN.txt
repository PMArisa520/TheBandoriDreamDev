	USN_fire_intensity = {
		if = {
			limit = {
				has_idea = USN_big_fire_idea4
			}
			swap_ideas = {
				remove_idea = USN_big_fire_idea4
				add_idea = USN_big_fire_idea3
			}
		}
		else_if = {
			limit = { 
				has_idea = USN_big_fire_idea3
			}
			swap_ideas = {
				remove_idea = USN_big_fire_idea3
				add_idea = USN_big_fire_idea2
			}
		}
		else_if = {
			limit = {
				has_idea = USN_big_fire_idea2
			}
			swap_ideas = {
				remove_idea = USN_big_fire_idea2
				add_idea = USN_big_fire_idea1
			}
		}
		else_if = {
			limit = { 
				has_idea = USN_big_fire_idea1 
			}
			swap_ideas = {
				remove_idea = USN_big_fire_idea1
				add_idea = USN_big_fire_idea
			}
		}
		else_if = {
			limit = { 
				has_idea = USN_big_fire_idea
			}
			add_stability = -0.05
		}
		else = {
			add_ideas = USN_big_fire_idea4
		}
	}
